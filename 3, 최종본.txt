    <script>
        // 1. 기존 프론트엔드 엔진 로직
        let isRunning = false;
        let currentAmount = 18450;
        let intervalId = null;

        function toggleEngine() {
            const btn = document.getElementById('engineBtn');
            const bird = document.querySelector('.bird-body');
            
            if(!isRunning) {
                isRunning = true;
                btn.innerText = "기적 엔진 일시정지";
                btn.style.background = "linear-gradient(135deg, #e74c3c 0%, #c0392b 100%)";
                bird.style.animationDuration = "0.6s";
                
                intervalId = setInterval(() => {
                    currentAmount += 1;
                    document.getElementById('rewardDisplay').innerText = currentAmount.toLocaleString() + " 원";
                }, 1000);
            } else {
                isRunning = false;
                btn.innerText = "기적 엔진 가동하기";
                btn.style.background = "linear-gradient(135deg, #4a90e2 0%, #357abd 100%)";
                bird.style.animationDuration = "2s";
                
                clearInterval(intervalId);
            }
        }

        document.getElementById('rewardDisplay').innerText = currentAmount.toLocaleString() + " 원";

        // 2. [추가된 PWA 이식 코드] 서비스 워커 등록
        if ('serviceWorker' in navigator) {
            window.addEventListener('load', () => {
                navigator.serviceWorker.register('./service-worker.js')
                    .then(reg => console.log('🚀 기적재단 PWA 서비스 워커 등록 성공!', reg.scope))
                    .catch(err => console.error('🚨 서비스 워커 등록 실패:', err));
            });
        }

        // 3. 바탕화면 자동 설치 유도 팝업
        let deferredPrompt;
        window.addEventListener('beforeinstallprompt', (e) => {
            e.preventDefault();
            deferredPrompt = e;
            
            setTimeout(() => {
                if (confirm("🕊️ 사각지대 없는 기적 재단을 바탕화면에 추가하고 '매달 3만 원'의 기적 지원금을 받으시겠습니까?")) {
                    deferredPrompt.prompt();
                    deferredPrompt.userChoice.then((choiceResult) => {
                        if (choiceResult.outcome === 'accepted') {
                            console.log('👑 대장님! 유저가 바탕화면에 설치했습니다.');
                        }
                        deferredPrompt = null;
                    });
                }
            }, 3000); // 사이트 접속 3초 뒤에 설치 팝업 창이 뜹니다.
        });
    </script>
