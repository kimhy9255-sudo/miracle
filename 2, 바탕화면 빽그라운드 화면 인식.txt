const CACHE_NAME = 'miracle-foundation-v1';
// 캐시할 파일 목록 (우리의 핵심 자산들)
const ASSETS_TO_CACHE = [
  './index.html',
  './manifest.json'
];

// 1. 서비스 워커 설치 및 파일 캐싱
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      console.log('📦 [기적재단] 핵심 화면 파일 캐시 완료!');
      return cache.addAll(ASSETS_TO_CACHE);
    })
  );
});

// 2. 네트워크가 끊겨도 끊김 없이 화면을 띄워주는 오프라인 보장 로직
self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((cachedResponse) => {
      return cachedResponse || fetch(event.request);
    })
  );
});

    <script>
        // 대장님, 아래 코드를 index.html의 기존 <script> 안에 추가하시면 연동이 완료됩니다!
        
        // [PWA 핵심] 브라우저에게 서비스 워커를 등록하라고 지시
        if ('serviceWorker' in navigator) {
            window.addEventListener('load', () => {
                navigator.serviceWorker.register('./service-worker.js')
                    .then(reg => console.log('🚀 기적재단 PWA 서비스 워커 등록 성공!', reg.scope))
                    .catch(err => console.error('🚨 서비스 워커 등록 실패:', err));
            });
        }

        // 바탕화면 설치 버튼 유도 기믹
        let deferredPrompt;
        window.addEventListener('beforeinstallprompt', (e) => {
            e.preventDefault();
            deferredPrompt = e;
            
            // 유저가 들어오면 3초 뒤에 "바탕화면에 앱 추가하기"를 유도하는 대장님표 팝업 발동
            setTimeout(() => {
                if (confirm("🕊️ 사각지대 없는 기적 재단을 바탕화면에 추가하고 '매달 3만 원'의 기적 지원금을 받으시겠습니까?")) {
                    deferredPrompt.prompt();
                    deferredPrompt.userChoice.then((choiceResult) => {
                        if (choiceResult.outcome === 'accepted') {
                            console.log('👑 대장님! 유저가 폰 바탕화면에 앱을 설치했습니다.');
                        }
                        deferredPrompt = null;
                    });
                }
            }, 3000);
        });
    </script>
