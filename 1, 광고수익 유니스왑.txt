import requests
from flask import Flask, request, jsonify
from eth_account import Account
import secrets

app = Flask(__name__)

# [보안] 국가급 해킹 방어를 위한 유저 데이터베이스 (예시용 메모리 딕셔너리)
user_db = {}

@app.route('/api/v1/engine/sync', methods=['POST'])
def sync_engine_earnings():
    data = request.json
    user_id = data.get('user_id')
    watch_time_sec = data.get('watch_time') # 유저가 기적 엔진을 켜둔 시간 (초)
    
    # 1. 매크로 및 비정상 트래픽 필터링 (보안 장벽)
    if watch_time_sec > 86400: # 하루는 24시간(86400초)을 넘을 수 없음
        return jsonify({"status": "error", "message": "비정상적인 접근이 감지되었습니다."}), 400
        
    # 2. 광고 수익 정산 (초당 eCPM 가치 계산)
    # 예시: 1초 방치당 약 0.5원의 광고 가치 환산
    earned_krw = watch_time_sec * 0.5 
    
    if user_id not in user_db:
        # 유저가 처음 들어왔다면, 복잡한 가입 없이 백엔드에서 Web3 지갑 주소 자동 생성 (Web3Auth 레이어)
        priv_key = secrets.token_hex(32)
        acct = Account.from_key(priv_key)
        user_db[user_id] = {
            "balance_krw": 0,
            "wallet_address": acct.address,
            "private_key": priv_key # 실제 서비스 시엔 암호화하여 Vault에 안전 보관
        }
    
    # 수익 누적
    user_db[user_id]["balance_krw"] += earned_krw
    
    return jsonify({
        "status": "success",
        "current_balance": user_db[user_id]["balance_krw"],
        "assigned_wallet": user_db[user_id]["wallet_address"]
    })

if __name__ == '__main__':
    app.run(port=5000)
